31/// Chain of Thought reasoning agent
32pub struct ChainOfThoughtAgent {
33    config: AgentConfig,
34    tools: Arc<ToolRegistry>,
35    reasoning_history: Arc<RwLock<Vec<ReasoningStep>>>,
36}

38#[derive(Debug, Clone, Serialize, Deserialize)]
39pub struct ReasoningStep {
40    pub step_number: usize,
41    pub thought: String,
42    pub action: Option<String>,
43    pub observation: Option<String>,
44    pub conclusion: Option<String>,
45}
46
47impl ChainOfThoughtAgent {
48    pub fn new(config: AgentConfig, tools: Arc<ToolRegistry>) -> Self {
49        Self {
50            config,
51            tools,
52            reasoning_history: Arc::new(RwLock::new(Vec::new())),
53        }
54    }
55    
56    async fn think_step_by_step(
57        &self,
58        input: &str,
59        state: &StateData,
60    ) -> Result<Vec<ReasoningStep>> {
61        let mut steps = Vec::new();
62        let mut current_thought = input.to_string();
63        let max_steps = 10;
64        
65        for step_num in 1..=max_steps {
66            // Generate thought about the current situation
67            let thought = format!(
68                "Step {}: Analyzing '{}' with state context",
69                step_num, current_thought
70            );
71            
72            // Determine if we need to use a tool
73            let action = self.determine_action(&current_thought, state).await?;
74            
75            let mut step = ReasoningStep {
76                step_number: step_num,
77                thought: thought.clone(),
78                action: action.clone(),
79                observation: None,
80                conclusion: None,
81            };
82            
83            // Execute action if needed
84            if let Some(action_str) = action {
85                let observation = self.execute_action(&action_str, state).await?;
86                step.observation = Some(observation.clone());
87                
88                // Update current thought based on observation
89                current_thought = format!("Based on {}, {}", action_str, observation);
90            }
91            
92            // Check if we have enough information to conclude
93            if self.can_conclude(&steps, state).await? {
94                step.conclusion = Some(self.formulate_conclusion(&steps, state).await?);
95                steps.push(step);
96                break;
97            }
98            
99            steps.push(step);
100        }
101        
102        Ok(steps)
103    }
104    
105    async fn determine_action(&self, thought: &str, state: &StateData) -> Result<Option<String>> {
106        // Analyze thought to determine if tool use is needed
107        if thought.contains("calculate") || thought.contains("compute") {
108            return Ok(Some("use_calculator".to_string()));
109        }
110        
111        if thought.contains("search") || thought.contains("find") {
112            return Ok(Some("search_knowledge".to_string()));
113        }
114        
115        if thought.contains("transform") || thought.contains("manipulate") {
116            return Ok(Some("use_string_tool".to_string()));
117        }
118        
119        // Check state for hints about needed actions
120        if let Some(Value::String(task)) = state.get("task_type") {
121            match task.as_str() {
122                "calculation" => return Ok(Some("use_calculator".to_string())),
123                "data_processing" => return Ok(Some("process_data".to_string())),
124                "api_request" => return Ok(Some("make_api_call".to_string())),
125                _ => {}
126            }
127        }
128        
129        Ok(None)
130    }
131    
132    async fn execute_action(&self, action: &str, state: &StateData) -> Result<String> {
133        match action {
134            "use_calculator" => {
135                if let Some(tool) = self.tools.get("calculator") {
136                    let params = self.extract_calculator_params(state)?;
137                    let context = ToolContext {
138                        state: state.clone(),
139                        metadata: HashMap::new(),
140                        auth: None,
141                        timeout: Some(30),
142                    };
143                    
144                    let result = tool.execute(params, context).await?;
145                    if let Some(data) = result.data {
146                        return Ok(format!("Calculation result: {}", data));
147                    }
148                }
149                Ok("Calculator tool not available".to_string())
150            }
151            
152            "search_knowledge" => {
153                // Simulate knowledge search
154                if let Some(Value::String(query)) = state.get("query") {
155                    Ok(format!("Found information about: {}", query))
156                } else {
157                    Ok("No search query provided".to_string())
158                }
159            }
160            
161            "use_string_tool" => {
162                if let Some(tool) = self.tools.get("string_tool") {
163                    let params = self.extract_string_params(state)?;
164                    let context = ToolContext {
165                        state: state.clone(),
166                        metadata: HashMap::new(),
167                        auth: None,
168                        timeout: Some(30),
169                    };
170                    
171                    let result = tool.execute(params, context).await?;
172                    if let Some(data) = result.data {
173                        return Ok(format!("String operation result: {}", data));
174                    }
175                }
176                Ok("String tool not available".to_string())
177            }
178            
179            _ => Ok(format!("Executed action: {}", action)),
180        }
181    }
182    
183    fn extract_calculator_params(&self, state: &StateData) -> Result<Value> {
184        // Extract calculation parameters from state
185        let operation = state.get("operation")
186            .and_then(|v| v.as_str())
187            .unwrap_or("add");
188        
189        let operands = state.get("operands")
190            .and_then(|v| v.as_array())
191            .map(|arr| arr.iter().filter_map(|v| v.as_f64()).collect::<Vec<_>>())
192            .unwrap_or_else(|| vec![0.0]);
193        
194        Ok(json!({
195            "operation": operation,
196            "operands": operands
197        }))
198    }
199    
200    fn extract_string_params(&self, state: &StateData) -> Result<Value> {
201        // Extract string operation parameters from state
202        let operation = state.get("string_op")
203            .and_then(|v| v.as_str())
204            .unwrap_or("concat");
205        
206        let input = state.get("text")
207            .and_then(|v| v.as_str())
208            .unwrap_or("");
209        
210        Ok(json!({
211            "operation": operation,
212            "text": input
213        }))
214    }
215    
216    async fn can_conclude(&self, steps: &[ReasoningStep], state: &StateData) -> Result<bool> {
217        // Determine if we have enough information to conclude
218        if steps.is_empty() {
219            return Ok(false);
220        }
221        
222        // Check if we've gathered enough observations
223        let observations_count = steps.iter()
224            .filter(|s| s.observation.is_some())
225            .count();
226        
227        // Check if state indicates completion
228        if let Some(Value::Bool(complete)) = state.get("reasoning_complete") {
229            return Ok(*complete);
230        }
231        
232        // Conclude if we have sufficient observations or reached max steps
233        Ok(observations_count >= 2 || steps.len() >= 5)
234    }
235    
236    async fn formulate_conclusion(
237        &self,
238        steps: &[ReasoningStep],
239        state: &StateData,
240    ) -> Result<String> {
241        let mut conclusion = String::from("Based on my reasoning:\n");
242        
243        // Summarize key observations
244        for step in steps {
245            if let Some(obs) = &step.observation {
246                conclusion.push_str(&format!("- {}\n", obs));
247            }
248        }
249        
250        // Add final conclusion based on state
251        if let Some(Value::String(goal)) = state.get("goal") {
252            conclusion.push_str(&format!("\nConclusion: Achieved goal of {}", goal));
253        } else {
254            conclusion.push_str("\nConclusion: Completed reasoning process successfully");
255        }
256        
257        Ok(conclusion)
258    }
259}

261#[async_trait]
262impl Agent for ChainOfThoughtAgent {
263    fn config(&self) -> &AgentConfig {
264        &self.config
265    }
266    
267    async fn process(&self, input: String, state: StateData) -> Result<StateData> {
268        // Perform chain of thought reasoning
269        let reasoning_steps = self.think_step_by_step(&input, &state).await?;
270        
271        // Store reasoning history
272        {
273            let mut history = self.reasoning_history.write().await;
274            history.extend(reasoning_steps.clone());
275        }
276        
277        // Update state with reasoning results
278        let mut new_state = state.clone();
279        new_state.insert(
280            "reasoning_steps".to_string(),
281            json!(reasoning_steps),
282        );
283        
284        if let Some(conclusion) = reasoning_steps.last().and_then(|s| s.conclusion.clone()) {
285            new_state.insert("conclusion".to_string(), json!(conclusion));
286        }
287        
288        Ok(new_state)
289    }
290    
291    async fn chat(&self, conversation: Conversation, state: StateData) -> Result<Conversation> {
292        let mut new_conversation = conversation.clone();
293        
294        // Process the last user message
295        if let Some(last_message) = conversation.messages.last() {
296            if last_message.role == MessageRole::User {
297                let reasoning_steps = self.think_step_by_step(&last_message.content, &state).await?;
298                
299                // Add reasoning as assistant message
300                let mut response = String::from("Let me think through this step by step:\n\n");
301                
302                for step in &reasoning_steps {
303                    response.push_str(&format!("{}. {}\n", step.step_number, step.thought));
304                    
305                    if let Some(action) = &step.action {
306                        response.push_str(&format!("   Action: {}\n", action));
307                    }
308                    
309                    if let Some(obs) = &step.observation {
310                        response.push_str(&format!("   Observation: {}\n", obs));
311                    }
312                    
313                    if let Some(conclusion) = &step.conclusion {
314                        response.push_str(&format!("\n{}\n", conclusion));
315                    }
316                }
317                
318                new_conversation.messages.push(Message {
319                    role: MessageRole::Assistant,
320                    content: response,
321                    name: Some(self.config.name.clone()),
322                    metadata: None,
323                });
324            }
325        }
326        
327        Ok(new_conversation)
328    }

331/// ReAct (Reasoning and Acting) agent
332pub struct ReActAgent {
333    config: AgentConfig,
334    tools: Arc<ToolRegistry>,
335    max_iterations: usize,
336}
337
338impl ReActAgent {
339    pub fn new(config: AgentConfig, tools: Arc<ToolRegistry>) -> Self {
340        Self {
341            config,
342            tools,
343            max_iterations: 5,
344        }
345    }
346    
347    async fn reason_and_act(&self, task: &str, state: &StateData) -> Result<Vec<ReActStep>> {
348        let mut steps = Vec::new();
349        let mut current_context = task.to_string();
350        
351        for iteration in 1..=self.max_iterations {
352            // Reasoning phase
353            let thought = self.reason(&current_context, state).await?;
354            
355            // Acting phase
356            let action = self.decide_action(&thought, state).await?;
357            
358            let mut step = ReActStep {
359                iteration,
360                thought: thought.clone(),
361                action: action.clone(),
362                observation: None,
363            };
364            
365            // Execute action and observe
366            if let Some(action_spec) = action {
367                let observation = self.act(&action_spec, state).await?;
368                step.observation = Some(observation.clone());
369                
370                // Update context with observation
371                current_context = format!(
372                    "Previous: {}\nAction taken: {:?}\nResult: {}",
373                    current_context, action_spec, observation
374                );
375                
376                // Check if task is complete
377                if self.is_task_complete(&observation, state).await? {
378                    steps.push(step);
379                    break;
380                }
381            }
382            
383            steps.push(step);
384        }
385        
386        Ok(steps)
387    }
388    
389    async fn reason(&self, context: &str, state: &StateData) -> Result<String> {
390        // Generate reasoning based on context and state
391        let mut thought = format!("Analyzing: {}", context);
392        
393        // Consider state information
394        if let Some(Value::Object(constraints)) = state.get("constraints") {
395            thought.push_str("\nConsidering constraints: ");
396            for (key, value) in constraints {
397                thought.push_str(&format!("{}: {}, ", key, value));
398            }
399        }
400        
401        // Consider available tools
402        let available_tools = self.tools.list();
403        if !available_tools.is_empty() {
404            thought.push_str(&format!("\nAvailable tools: {:?}", available_tools));
405        }
406        
407        Ok(thought)
408    }
409    
410    async fn decide_action(&self, thought: &str, state: &StateData) -> Result<Option<ActionSpec>> {
411        // Decide on action based on reasoning
412        if thought.contains("calculate") || thought.contains("compute") {
413            return Ok(Some(ActionSpec {
414                tool: "calculator".to_string(),
415                parameters: self.extract_calculator_params(state)?,
416            }));
417        }
418        
419        if thought.contains("transform") || thought.contains("text") {
420            return Ok(Some(ActionSpec {
421                tool: "string_tool".to_string(),
422                parameters: self.extract_string_params(state)?,
423            }));
424        }
425        
426        if thought.contains("request") || thought.contains("API") {
427            return Ok(Some(ActionSpec {
428                tool: "http_client".to_string(),
429                parameters: json!({
430                    "method": "GET",
431                    "url": state.get("api_url").and_then(|v| v.as_str()).unwrap_or("https://api.example.com"),
432                }),
433            }));
434        }
435        
436        Ok(None)
437    }
438    
439    async fn act(&self, action: &ActionSpec, state: &StateData) -> Result<String> {
440        // Execute the action using tools
441        if let Some(tool) = self.tools.get(&action.tool) {
442            let context = ToolContext {
443                state: state.clone(),
444                metadata: HashMap::new(),
445                auth: None,
446                timeout: Some(30),
447            };
448            
449            let result = tool.execute(action.parameters.clone(), context).await?;
450            
451            if result.success {
452                if let Some(data) = result.data {
453                    return Ok(format!("Success: {}", data));
454                }
455                return Ok("Action completed successfully".to_string());
456            } else {
457                return Ok(format!(
458                    "Action failed: {}",
459                    result.error.unwrap_or_else(|| "Unknown error".to_string())
460                ));
461            }
462        }
463        
464        Ok(format!("Tool '{}' not available", action.tool))
465    }
466    
467    async fn is_task_complete(&self, observation: &str, state: &StateData) -> Result<bool> {
468        // Check if the task is complete based on observation
469        if observation.contains("Success") || observation.contains("completed") {
470            return Ok(true);
471        }
472        
473        // Check state for completion criteria
474        if let Some(Value::Bool(complete)) = state.get("task_complete") {
475            return Ok(*complete);
476        }
477        
478        if let Some(Value::String(status)) = state.get("status") {
479            return Ok(status == "complete" || status == "done");
480        }
481        
482        Ok(false)
483    }
484    
485    fn extract_calculator_params(&self, state: &StateData) -> Result<Value> {
486        let operation = state.get("operation")
487            .and_then(|v| v.as_str())
488            .unwrap_or("add");
489        
490        let operands = state.get("operands")
491            .and_then(|v| v.as_array())
492            .map(|arr| arr.iter().filter_map(|v| v.as_f64()).collect::<Vec<_>>())
493            .unwrap_or_else(|| vec![1.0, 1.0]);
494        
495        Ok(json!({
496            "operation": operation,
497            "operands": operands
498        }))
499    }
500    
501    fn extract_string_params(&self, state: &StateData) -> Result<Value> {
502        let operation = state.get("string_op")
503            .and_then(|v| v.as_str())
504            .unwrap_or("concat");
505        
506        let text = state.get("text")
507            .and_then(|v| v.as_str())
508            .unwrap_or("default text");
509        
510        Ok(json!({
511            "operation": operation,
512            "text": text,
513            "separator": state.get("separator").and_then(|v| v.as_str()).unwrap_or(" "),
514        }))
515    }
516}

591/// Memory-enhanced agent with short-term and long-term memory
592pub struct MemoryAgent {
593    config: AgentConfig,
594    short_term_memory: Arc<RwLock<Vec<MemoryItem>>>,
595    long_term_memory: Arc<RwLock<HashMap<String, MemoryItem>>>,
596    working_memory_limit: usize,
597}
598
599#[derive(Debug, Clone, Serialize, Deserialize)]
600pub struct MemoryItem {
601    pub id: String,
602    pub content: String,
603    pub importance: f32,
604    pub timestamp: u64,
605    pub access_count: usize,
606    pub associations: Vec<String>,
607}
608
609impl MemoryAgent {
610    pub fn new(config: AgentConfig) -> Self {
611        Self {
612            config,
613            short_term_memory: Arc::new(RwLock::new(Vec::new())),
614            long_term_memory: Arc::new(RwLock::new(HashMap::new())),
615            working_memory_limit: 7, // Miller's Law
616        }
617    }
618    
619    async fn store_memory(&self, content: String, importance: f32) -> Result<String> {
620        let id = format!("mem_{}", uuid::Uuid::new_v4());
621        let memory = MemoryItem {
622            id: id.clone(),
623            content,
624            importance,
625            timestamp: std::time::SystemTime::now()
626                .duration_since(std::time::UNIX_EPOCH)
627                .unwrap()
628                .as_secs(),
629            access_count: 0,
630            associations: Vec::new(),
631        };
632        
633        // Add to short-term memory
634        {
635            let mut stm = self.short_term_memory.write().await;
636            stm.push(memory.clone());
637            
638            // Maintain working memory limit
639            if stm.len() > self.working_memory_limit {
640                // Move least important to long-term memory
641                stm.sort_by(|a, b| b.importance.partial_cmp(&a.importance).unwrap());
642                let to_archive = stm.split_off(self.working_memory_limit);
643                
644                let mut ltm = self.long_term_memory.write().await;
645                for item in to_archive {
646                    ltm.insert(item.id.clone(), item);
647                }
648            }
649        }
650        
651        Ok(id)
652    }
653    
654    async fn recall_memory(&self, query: &str) -> Result<Vec<MemoryItem>> {
655        let mut relevant_memories = Vec::new();
656        
657        // Search short-term memory
658        {
659            let mut stm = self.short_term_memory.write().await;
660            for memory in stm.iter_mut() {
661                if memory.content.contains(query) {
662                    memory.access_count += 1;
663                    relevant_memories.push(memory.clone());
664                }
665            }
666        }
667        
668        // Search long-term memory
669        {
670            let mut ltm = self.long_term_memory.write().await;
671            for (_, memory) in ltm.iter_mut() {
672                if memory.content.contains(query) {
673                    memory.access_count += 1;
674                    memory.importance *= 1.1; // Increase importance on recall
675                    relevant_memories.push(memory.clone());
676                }
677            }
678        }
679        
680        // Sort by relevance (importance * access_count)
681        relevant_memories.sort_by(|a, b| {
682            let score_a = a.importance * a.access_count as f32;
683            let score_b = b.importance * b.access_count as f32;
684            score_b.partial_cmp(&score_a).unwrap()
685        });
686        
687        Ok(relevant_memories)
688    }
689    
690    async fn consolidate_memories(&self) -> Result<()> {
691        // Consolidate related memories
692        let stm = self.short_term_memory.read().await;
693        let memories_to_consolidate: Vec<_> = stm.iter()
694            .filter(|m| m.access_count > 3)
695            .cloned()
696            .collect();
697        drop(stm);
698        
699        if memories_to_consolidate.len() >= 2 {
700            // Create associations between frequently accessed memories
701            let mut ltm = self.long_term_memory.write().await;
702            for i in 0..memories_to_consolidate.len() {
703                for j in i+1..memories_to_consolidate.len() {
704                    let mem1 = &memories_to_consolidate[i];
705                    let mem2 = &memories_to_consolidate[j];
706                    
707                    // Simple similarity check
708                    if Self::calculate_similarity(&mem1.content, &mem2.content) > 0.3 {
709                        if let Some(stored_mem1) = ltm.get_mut(&mem1.id) {
710                            stored_mem1.associations.push(mem2.id.clone());
711                        }
712                        if let Some(stored_mem2) = ltm.get_mut(&mem2.id) {
713                            stored_mem2.associations.push(mem1.id.clone());
714                        }
715                    }
716                }
717            }
718        }
719        
720        Ok(())
721    }
722    
723    fn calculate_similarity(text1: &str, text2: &str) -> f32 {
724        // Simple word overlap similarity
725        let words1: std::collections::HashSet<_> = text1.split_whitespace().collect();
726        let words2: std::collections::HashSet<_> = text2.split_whitespace().collect();
727        
728        let intersection = words1.intersection(&words2).count() as f32;
729        let union = words1.union(&words2).count() as f32;
730        
731        if union > 0.0 {
732            intersection / union
733        } else {
734            0.0
735        }
736    }
737}

739#[async_trait]
740impl Agent for MemoryAgent {
741    fn config(&self) -> &AgentConfig {
742        &self.config
743    }
744    
745    async fn process(&self, input: String, state: StateData) -> Result<StateData> {
746        // Store input as memory
747        let importance = state.get("importance")
748            .and_then(|v| v.as_f64())
749            .unwrap_or(0.5) as f32;
750        
751        let memory_id = self.store_memory(input.clone(), importance).await?;
752        
753        // Recall relevant memories
754        let query = state.get("recall_query")
755            .and_then(|v| v.as_str())
756            .unwrap_or(&input);
757        
758        let recalled = self.recall_memory(query).await?;
759        
760        // Consolidate memories periodically
761        self.consolidate_memories().await?;
762        
763        // Update state with memory information
764        let mut new_state = state.clone();
765        new_state.insert("stored_memory_id".to_string(), json!(memory_id));
766        new_state.insert("recalled_memories".to_string(), json!(recalled));
767        
768        // Add memory statistics
769        let stm_count = self.short_term_memory.read().await.len();
770        let ltm_count = self.long_term_memory.read().await.len();
771        
772        new_state.insert("memory_stats".to_string(), json!({
773            "short_term_count": stm_count,
774            "long_term_count": ltm_count,
775            "working_memory_usage": format!("{}/{}", stm_count, self.working_memory_limit),
776        }));
777        
778        Ok(new_state)
779    }
780    
781    async fn chat(&self, conversation: Conversation, _state: StateData) -> Result<Conversation> {
782        let mut new_conversation = conversation.clone();
783        
784        if let Some(last_message) = conversation.messages.last() {
785            if last_message.role == MessageRole::User {
786                // Store the message
787                self.store_memory(last_message.content.clone(), 0.7).await?;
788                
789                // Recall relevant context
790                let memories = self.recall_memory(&last_message.content).await?;
791                
792                let mut response = String::from("Based on my memory:\n\n");
793                
794                if !memories.is_empty() {
795                    response.push_str("Relevant context:\n");
796                    for (i, memory) in memories.iter().take(3).enumerate() {
797                        response.push_str(&format!(
798                            "{}. {} (importance: {:.2}, accessed: {} times)\n",
799                            i + 1,
800                            memory.content,
801                            memory.importance,
802                            memory.access_count
803                        ));
804                    }
805                    response.push_str("\n");
806                }
807                
808                response.push_str(&format!(
809                    "Response: I've stored your message and recalled {} relevant memories.",
810                    memories.len()
811                ));
812                
813                new_conversation.messages.push(Message {
814                    role: MessageRole::Assistant,
815                    content: response,
816                    name: Some(self.config.name.clone()),
817                    metadata: None,
818                });
819            }
820        }
821        
822        Ok(new_conversation)
823    }
824}

826// UUID generation helper (simplified)
827mod uuid {
828    use std::sync::atomic::{AtomicU64, Ordering};
829    
830    static COUNTER: AtomicU64 = AtomicU64::new(0);
831    
832    pub struct Uuid;
833    
834    impl Uuid {
835        pub fn new_v4() -> String {
836            let count = COUNTER.fetch_add(1, Ordering::SeqCst);
837            let timestamp = std::time::SystemTime::now()
838                .duration_since(std::time::UNIX_EPOCH)
839                .unwrap()
840                .as_nanos();
841            format!("{:x}-{:x}", timestamp, count)
842        }
843    }
844}

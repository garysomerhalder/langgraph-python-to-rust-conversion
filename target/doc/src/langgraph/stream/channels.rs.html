20impl ChannelRegistry {
21    pub fn new() -> Self {
22        Self {
23            channels: Arc::new(RwLock::new(HashMap::new())),
24        }
25    }
26    
27    pub async fn create_channel(
28        &self,
29        name: String,
30        channel_type: ChannelType,
31    ) -> Result<(), GraphError> {
32        let mut channels = self.channels.write().await;
33        
34        if channels.contains_key(&name) {
35            return Err(GraphError::EdgeError(
36                format!("Channel {} already exists", name)
37            ));
38        }
39        
40        let endpoint = match channel_type {
41            ChannelType::Broadcast(capacity) => {
42                let (tx, _) = broadcast::channel(capacity);
43                ChannelEndpoint::Broadcast(Arc::new(tx))
44            }
45            ChannelType::Mpsc(capacity) => {
46                let (tx, rx) = mpsc::channel(capacity);
47                ChannelEndpoint::Mpsc {
48                    sender: Arc::new(tx),
49                    receiver: Arc::new(RwLock::new(Some(rx))),
50                }
51            }
52            ChannelType::Oneshot => {
53                ChannelEndpoint::Oneshot {
54                    pending: Arc::new(RwLock::new(Vec::new())),
55                }
56            }
57        };
58        
59        channels.insert(name, endpoint);
60        Ok(())
61    }

123impl<T: Serialize + Clone + Send + Sync> ChannelSender<T> {
124    
125    pub async fn send(&self, value: T) -> Result<(), GraphError> {
126        let bytes = bincode::serialize(&value)
127            .map_err(|e| GraphError::SerializationError(e.to_string()))?;
128        
129        match &self.endpoint {
130            ChannelEndpoint::Broadcast(tx) => {
131                tx.send(bytes)
132                    .map_err(|_| GraphError::EdgeError("Failed to send on broadcast channel".to_string()))?;
133                Ok(())
134            }
135            ChannelEndpoint::Mpsc { sender, .. } => {
136                sender.send(bytes).await
137                    .map_err(|_| GraphError::EdgeError("Failed to send on mpsc channel".to_string()))?;
138                Ok(())
139            }
140            ChannelEndpoint::Oneshot { pending } => {
141                let mut pending_lock = pending.write().await;
142                if let Some(tx) = pending_lock.pop() {
143                    tx.send(bytes)
144                        .map_err(|_| GraphError::EdgeError("Failed to send on oneshot channel".to_string()))?;
145                    Ok(())
146                } else {
147                    Err(GraphError::EdgeError("No receiver for oneshot channel".to_string()))
148                }
149            }
150        }
151    }
152}

170impl<T: for<'de> Deserialize<'de> + Clone + Send + Sync> ChannelReceiver<T> {
171    
172    pub async fn recv(&mut self) -> Result<T, GraphError> {
173        match &self.endpoint {
174            ChannelEndpoint::Broadcast(tx) => {
175                let mut rx = tx.subscribe();
176                let bytes = rx.recv().await
177                    .map_err(|_| GraphError::EdgeError("Failed to receive from broadcast channel".to_string()))?;
178                
179                bincode::deserialize(&bytes)
180                    .map_err(|e| GraphError::SerializationError(e.to_string()))
181            }
182            ChannelEndpoint::Mpsc { receiver, .. } => {
183                let mut receiver_lock = receiver.write().await;
184                if let Some(ref mut rx) = *receiver_lock {
185                    let bytes = rx.recv().await
186                        .ok_or_else(|| GraphError::EdgeError("Channel closed".to_string()))?;
187                    
188                    bincode::deserialize(&bytes)
189                        .map_err(|e| GraphError::SerializationError(e.to_string()))
190                } else {
191                    Err(GraphError::EdgeError("Receiver already taken".to_string()))
192                }
193            }
194            ChannelEndpoint::Oneshot { pending } => {
195                let (tx, rx) = oneshot::channel();
196                pending.write().await.push(tx);
197                
198                let bytes = rx.await
199                    .map_err(|_| GraphError::EdgeError("Failed to receive from oneshot channel".to_string()))?;
200                
201                bincode::deserialize(&bytes)
202                    .map_err(|e| GraphError::SerializationError(e.to_string()))
203            }
204        }
205    }
206}

225#[async_trait]
226impl ChannelSelector for RoundRobinSelector {
227    async fn select(&self, channels: Vec<String>) -> Result<String, GraphError> {
228        if channels.is_empty() {
229            return Err(GraphError::EdgeError("No channels to select from".to_string()));
230        }
231        
232        let mut index = self.index.write().await;
233        let selected = channels[*index % channels.len()].clone();
234        *index += 1;
235        
236        Ok(selected)
237    }

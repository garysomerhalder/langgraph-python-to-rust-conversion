3use std::sync::Arc;
4use crate::state::StateData;
5use crate::graph::GraphError;
6use crate::Result;

8/// Type alias for conditional functions
9pub type ConditionFn = Arc<dyn Fn(&StateData) -> bool + Send + Sync>;
10
11/// Type alias for routing functions that determine next node
12pub type RoutingFn = Arc<dyn Fn(&StateData) -> String + Send + Sync>;
13
14/// A conditional edge that routes based on state evaluation
15#[derive(Clone)]
16pub struct ConditionalEdge {
17    /// Source node ID
18    pub from: String,
19    
20    /// Possible target nodes with their conditions
21    pub branches: Vec<ConditionalBranch>,
22    
23    /// Default target if no conditions match
24    pub default: Option<String>,
25}
26
27/// A single branch in a conditional edge
28#[derive(Clone)]
29pub struct ConditionalBranch {
30    /// Target node ID
31    pub target: String,
32    
33    /// Condition that must be true to take this branch
34    pub condition: ConditionFn,
35    
36    /// Optional priority (higher = evaluated first)
37    pub priority: Option<i32>,
38    
39    /// Optional metadata
40    pub metadata: Option<serde_json::Value>,
41}
42
43impl ConditionalEdge {
44    /// Create a new conditional edge
45    pub fn new(from: impl Into<String>) -> Self {
46        Self {
47            from: from.into(),
48            branches: Vec::new(),
49            default: None,
50        }
51    }
52    
53    /// Add a conditional branch
54    pub fn add_branch(
55        mut self,
56        to: impl Into<String>,
57        condition: ConditionFn,
58        priority: i32,
59    ) -> Self {
60        self.branches.push(ConditionalBranch {
61            target: to.into(),
62            condition,
63            priority: Some(priority),
64            metadata: None,
65        });
66        // Sort branches by priority (highest first)
67        self.branches.sort_by(|a, b| {
68            let p_a = a.priority.unwrap_or(0);
69            let p_b = b.priority.unwrap_or(0);
70            p_b.cmp(&p_a)
71        });
72        self
73    }
74    
75    /// Set the default target
76    pub fn with_default(mut self, default: impl Into<String>) -> Self {
77        self.default = Some(default.into());
78        self
79    }
80    
81    /// Evaluate the condition and return the target node
82    pub fn evaluate(&self, state: &StateData) -> Result<String> {
83        // Check each branch in priority order
84        for branch in &self.branches {
85            if (branch.condition)(state) {
86                return Ok(branch.target.clone());
87            }
88        }
89        
90        // Use default if no conditions matched
91        self.default
92            .clone()
93            .ok_or_else(|| GraphError::EdgeError(
94                format!("No matching condition for edge from {}", self.from)
95            ).into())
96    }
97}

99/// Builder for conditional routing
100pub struct ConditionalRouter {
101    /// Source node for this router
102    pub from: String,
103    edges: Vec<ConditionalEdge>,
104}
105
106impl ConditionalRouter {
107    /// Create a new conditional router
108    pub fn new(from: impl Into<String>) -> Self {
109        Self {
110            from: from.into(),
111            edges: Vec::new(),
112        }
113    }
114    
115    /// Add a conditional edge
116    pub fn add_edge(mut self, edge: ConditionalEdge) -> Self {
117        self.edges.push(edge);
118        self
119    }
120    
121    /// Route from a given node based on state
122    pub fn route(&self, from: &str, state: &StateData) -> Result<String> {
123        self.edges
124            .iter()
125            .find(|e| e.from == from)
126            .ok_or_else(|| GraphError::EdgeError(
127                format!("No conditional edge from {}", from)
128            ).into())
129            .and_then(|edge| edge.evaluate(state))
130    }
131    
132    /// Route with specific branches (for testing)
133    pub fn route_with_branches(&self, state: &StateData, branches: Vec<ConditionalBranch>) -> Option<String> {
134        for branch in branches {
135            if (branch.condition)(state) {
136                return Some(branch.target);
137            }
138        }
139        None
140    }
141}

143/// Advanced routing with multiple conditions
144pub struct MultiConditionalRouter {
145    /// Routing function that returns next node based on state
146    router: RoutingFn,
147    
148    /// Valid target nodes
149    valid_targets: Vec<String>,
150}
151
152impl MultiConditionalRouter {
153    /// Create a new multi-conditional router
154    pub fn new(router: RoutingFn, valid_targets: Vec<String>) -> Self {
155        Self {
156            router,
157            valid_targets,
158        }
159    }
160    
161    /// Route to next node
162    pub fn route(&self, state: &StateData) -> Result<String> {
163        let target = (self.router)(state);
164        
165        if self.valid_targets.contains(&target) {
166            Ok(target)
167        } else {
168            Err(GraphError::EdgeError(
169                format!("Invalid routing target: {}", target)
170            ).into())
171        }
172    }
173}

178    use serde_json::json;

180    #[test]
181    fn test_conditional_edge() {
182        let mut state = StateData::new();
183        state.insert("score".to_string(), json!(75));
184        
185        let edge = ConditionalEdge::new("start")
186            .add_branch(
187                "high_score",
188                Arc::new(|s| {
189                    s.get("score")
190                        .and_then(|v| v.as_i64())
191                        .map(|score| score >= 80)
192                        .unwrap_or(false)
193                }),
194                10,
195            )
196            .add_branch(
197                "medium_score",
198                Arc::new(|s| {
199                    s.get("score")
200                        .and_then(|v| v.as_i64())
201                        .map(|score| score >= 60 && score < 80)
202                        .unwrap_or(false)
203                }),
204                5,
205            )
206            .with_default("low_score");
207        
208        let target = edge.evaluate(&state).unwrap();
209        assert_eq!(target, "medium_score");
210    }

3use serde_json::Value;
4use crate::state::StateData;

7/// Condition evaluator for routing decisions
8pub struct ConditionEvaluator;
9
10impl ConditionEvaluator {
11    /// Evaluate a condition string against state
12    pub fn evaluate(condition: &str, state: &StateData) -> Result<bool> {
13        // Parse different condition types
14        if condition.starts_with("eq:") {
15            Self::evaluate_equality(condition, state)
16        } else if condition.starts_with("gt:") {
17            Self::evaluate_greater_than(condition, state)
18        } else if condition.starts_with("lt:") {
19            Self::evaluate_less_than(condition, state)
20        } else if condition.starts_with("contains:") {
21            Self::evaluate_contains(condition, state)
22        } else if condition.starts_with("exists:") {
23            Self::evaluate_exists(condition, state)
24        } else if condition.starts_with("fn:") {
25            Self::evaluate_function(condition, state)
26        } else {
27            // Default: check if value is truthy
28            Self::evaluate_truthy(condition, state)
29        }
30    }
31    
32    /// Evaluate equality condition (eq:field=value)
33    fn evaluate_equality(condition: &str, state: &StateData) -> Result<bool> {
34        let parts: Vec<&str> = condition[3..].split('=').collect();
35        if parts.len() != 2 {
36            return Ok(false);
37        }
38        
39        let field = parts[0];
40        let expected = parts[1];
41        
42        if let Some(value) = state.get(field) {
43            match value {
44                Value::String(s) => Ok(s == expected),
45                Value::Number(n) => {
46                    if let Ok(expected_num) = expected.parse::<f64>() {
47                        Ok(n.as_f64() == Some(expected_num))
48                    } else {
49                        Ok(false)
50                    }
51                }
52                Value::Bool(b) => {
53                    if let Ok(expected_bool) = expected.parse::<bool>() {
54                        Ok(*b == expected_bool)
55                    } else {
56                        Ok(false)
57                    }
58                }
59                _ => Ok(false),
60            }
61        } else {
62            Ok(false)
63        }
64    }
65    
66    /// Evaluate greater than condition (gt:field>value)
67    fn evaluate_greater_than(condition: &str, state: &StateData) -> Result<bool> {
68        let parts: Vec<&str> = condition[3..].split('>').collect();
69        if parts.len() != 2 {
70            return Ok(false);
71        }
72        
73        let field = parts[0];
74        let threshold = parts[1];
75        
76        if let Some(Value::Number(n)) = state.get(field) {
77            if let Ok(threshold_num) = threshold.parse::<f64>() {
78                Ok(n.as_f64().unwrap_or(0.0) > threshold_num)
79            } else {
80                Ok(false)
81            }
82        } else {
83            Ok(false)
84        }
85    }
86    
87    /// Evaluate less than condition (lt:field<value)
88    fn evaluate_less_than(condition: &str, state: &StateData) -> Result<bool> {
89        let parts: Vec<&str> = condition[3..].split('<').collect();
90        if parts.len() != 2 {
91            return Ok(false);
92        }
93        
94        let field = parts[0];
95        let threshold = parts[1];
96        
97        if let Some(Value::Number(n)) = state.get(field) {
98            if let Ok(threshold_num) = threshold.parse::<f64>() {
99                Ok(n.as_f64().unwrap_or(0.0) < threshold_num)
100            } else {
101                Ok(false)
102            }
103        } else {
104            Ok(false)
105        }
106    }
107    
108    /// Evaluate contains condition (contains:field:value)
109    fn evaluate_contains(condition: &str, state: &StateData) -> Result<bool> {
110        let parts: Vec<&str> = condition[9..].split(':').collect();
111        if parts.len() != 2 {
112            return Ok(false);
113        }
114        
115        let field = parts[0];
116        let search = parts[1];
117        
118        if let Some(value) = state.get(field) {
119            match value {
120                Value::String(s) => Ok(s.contains(search)),
121                Value::Array(arr) => {
122                    Ok(arr.iter().any(|v| {
123                        if let Value::String(s) = v {
124                            s == search
125                        } else {
126                            false
127                        }
128                    }))
129                }
130                _ => Ok(false),
131            }
132        } else {
133            Ok(false)
134        }
135    }
136    
137    /// Evaluate exists condition (exists:field)
138    fn evaluate_exists(condition: &str, state: &StateData) -> Result<bool> {
139        let field = &condition[7..];
140        Ok(state.contains_key(field))
141    }
142    
143    /// Evaluate function-based condition (fn:function_name)
144    fn evaluate_function(condition: &str, state: &StateData) -> Result<bool> {
145        let function_name = &condition[3..];
146        
147        match function_name {
148            "has_errors" => {
149                Ok(state.get("errors").and_then(|v| v.as_array()).map(|a| !a.is_empty()).unwrap_or(false))
150            }
151            "is_complete" => {
152                Ok(state.get("status").and_then(|v| v.as_str()).map(|s| s == "complete").unwrap_or(false))
153            }
154            "needs_review" => {
155                Ok(state.get("confidence").and_then(|v| v.as_f64()).map(|c| c < 0.8).unwrap_or(true))
156            }
157            _ => Ok(false),
158        }
159    }
160    
161    /// Evaluate if a field is truthy
162    fn evaluate_truthy(field: &str, state: &StateData) -> Result<bool> {
163        if let Some(value) = state.get(field) {
164            match value {
165                Value::Bool(b) => Ok(*b),
166                Value::Number(n) => Ok(n.as_f64().unwrap_or(0.0) != 0.0),
167                Value::String(s) => Ok(!s.is_empty()),
168                Value::Array(a) => Ok(!a.is_empty()),
169                Value::Object(o) => Ok(!o.is_empty()),
170                Value::Null => Ok(false),
171            }
172        } else {
173            Ok(false)
174        }
175    }
176    
177    /// Evaluate multiple conditions with logical operators
178    pub fn evaluate_complex(conditions: &[String], operator: &str, state: &StateData) -> Result<bool> {
179        match operator {
180            "AND" | "&&" => {
181                for condition in conditions {
182                    if !Self::evaluate(condition, state)? {
183                        return Ok(false);
184                    }
185                }
186                Ok(true)
187            }
188            "OR" | "||" => {
189                for condition in conditions {
190                    if Self::evaluate(condition, state)? {
191                        return Ok(true);
192                    }
193                }
194                Ok(false)
195            }
196            "XOR" => {
197                let mut true_count = 0;
198                for condition in conditions {
199                    if Self::evaluate(condition, state)? {
200                        true_count += 1;
201                    }
202                }
203                Ok(true_count == 1)
204            }
205            _ => Ok(false),
206        }
207    }
208}

210/// Router for selecting next nodes based on conditions
211pub struct ConditionalRouter {
212    routes: Vec<ConditionalRoute>,
213}
214
215#[derive(Debug, Clone)]
216pub struct ConditionalRoute {
217    pub condition: String,
218    pub target: String,
219    pub priority: i32,
220}
221
222impl ConditionalRouter {
223    /// Create a new conditional router
224    pub fn new() -> Self {
225        Self {
226            routes: Vec::new(),
227        }
228    }
229    
230    /// Add a conditional route
231    pub fn add_route(&mut self, condition: String, target: String, priority: i32) {
232        self.routes.push(ConditionalRoute {
233            condition,
234            target,
235            priority,
236        });
237    }
238    
239    /// Route based on state
240    pub fn route(&self, state: &StateData) -> Result<Option<String>> {
241        let mut matching_routes = Vec::new();
242        
243        for route in &self.routes {
244            if ConditionEvaluator::evaluate(&route.condition, state)? {
245                matching_routes.push(route);
246            }
247        }
248        
249        if matching_routes.is_empty() {
250            return Ok(None);
251        }
252        
253        // Sort by priority (higher priority first)
254        matching_routes.sort_by(|a, b| b.priority.cmp(&a.priority));
255        
256        Ok(Some(matching_routes[0].target.clone()))
257    }
258    
259    /// Get all matching routes (for parallel execution)
260    pub fn route_all(&self, state: &StateData) -> Result<Vec<String>> {
261        let mut targets = Vec::new();
262        
263        for route in &self.routes {
264            if ConditionEvaluator::evaluate(&route.condition, state)? {
265                targets.push(route.target.clone());
266            }
267        }
268        
269        Ok(targets)
270    }
271}

1use std::collections::HashMap;
2use tokio::sync::mpsc;
3
4#[derive(Debug)]
5pub struct StateChannels<T> 
6where 
7    T: Clone + Send + Sync + 'static
8{
9    channels: HashMap<String, mpsc::Sender<T>>,
10    receivers: HashMap<String, mpsc::Receiver<T>>,
11}
12
13impl<T> StateChannels<T>
14where 
15    T: Clone + Send + Sync + 'static
16{
17    pub fn new() -> Self {
18        Self {
19            channels: HashMap::new(),
20            receivers: HashMap::new(),
21        }
22    }
23
24    pub fn create_channel(&mut self, name: String, buffer_size: usize) {
25        let (tx, rx) = mpsc::channel(buffer_size);
26        self.channels.insert(name.clone(), tx);
27        self.receivers.insert(name, rx);
28    }
29
30    pub async fn send(&self, channel_name: &str, value: T) -> Result<(), mpsc::error::SendError<T>> {
31        if let Some(sender) = self.channels.get(channel_name) {
32            sender.send(value).await
33        } else {
34            Err(mpsc::error::SendError(value))
35        }
36    }
37
38    pub async fn receive(&mut self, channel_name: &str) -> Option<T> {
39        if let Some(receiver) = self.receivers.get_mut(channel_name) {
40            receiver.recv().await
41        } else {
42            None
43        }
44    }
45
46    pub fn get_sender(&self, channel_name: &str) -> Option<mpsc::Sender<T>> {
47        self.channels.get(channel_name).cloned()
48    }
49}
50
51impl<T> Clone for StateChannels<T>
52where 
53    T: Clone + Send + Sync + 'static
54{
55    fn clone(&self) -> Self {
56        Self {
57            channels: self.channels.clone(),
58            receivers: HashMap::new(), // Can't clone receivers
59        }
60    }
61}

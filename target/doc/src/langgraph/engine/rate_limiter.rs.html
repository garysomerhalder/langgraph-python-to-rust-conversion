1//! Rate limiting for graph execution
2//!
3//! Provides rate limiting capabilities to prevent resource exhaustion
4//! and ensure fair resource usage across multiple graph executions.
5
6use std::sync::Arc;
7use std::time::{Duration, Instant};
8use tokio::sync::{Mutex, Semaphore};
9use tokio::time;
10use thiserror::Error;
11
12/// Errors that can occur during rate limiting
13#[derive(Error, Debug, Clone)]
14pub enum RateLimitError {
15    #[error("Rate limit exceeded: {0}")]
16    RateLimitExceeded(String),
17    
18    #[error("Timeout waiting for rate limit permit")]
19    Timeout,
20    
21    #[error("Rate limiter is shutting down")]
22    ShuttingDown,
23    
24    #[error("Semaphore has been closed")]
25    SemaphoreClosed,
26}
27
28/// Rate limiter for controlling execution frequency
29pub struct RateLimiter {
30    /// Maximum number of executions per window
31    max_executions: usize,
32    
33    /// Time window for rate limiting
34    window: Duration,
35    
36    /// Semaphore for controlling concurrent access
37    semaphore: Arc<Semaphore>,
38    
39    /// Token bucket for rate limiting
40    token_bucket: Arc<Mutex<TokenBucket>>,
41}
42
43/// Token bucket for rate limiting algorithm
44struct TokenBucket {
45    /// Current number of tokens
46    tokens: f64,
47    
48    /// Maximum tokens in the bucket
49    capacity: f64,
50    
51    /// Refill rate (tokens per second)
52    refill_rate: f64,
53    
54    /// Last refill timestamp
55    last_refill: Instant,
56}
57
58impl TokenBucket {
59    /// Create a new token bucket
60    fn new(capacity: f64, refill_rate: f64) -> Self {
61        Self {
62            tokens: capacity,
63            capacity,
64            refill_rate,
65            last_refill: Instant::now(),
66        }
67    }
68    
69    /// Try to consume tokens
70    #[inline]
71    fn try_consume(&mut self, tokens: f64) -> bool {
72        self.refill();
73        
74        if self.tokens >= tokens {
75            self.tokens -= tokens;
76            true
77        } else {
78            false
79        }
80    }
81    
82    /// Refill tokens based on elapsed time
83    #[inline]
84    fn refill(&mut self) {
85        let now = Instant::now();
86        let elapsed = now.duration_since(self.last_refill);
87        let tokens_to_add = elapsed.as_secs_f64() * self.refill_rate;
88        
89        self.tokens = (self.tokens + tokens_to_add).min(self.capacity);
90        self.last_refill = now;
91    }
92    
93    /// Calculate wait time for required tokens
94    fn wait_time(&self, tokens: f64) -> Duration {
95        if self.tokens >= tokens {
96            return Duration::ZERO;
97        }
98        
99        let tokens_needed = tokens - self.tokens;
100        let seconds_to_wait = tokens_needed / self.refill_rate;
101        Duration::from_secs_f64(seconds_to_wait)
102    }
103}
104
105impl RateLimiter {
106    /// Create a new rate limiter
107    pub fn new(max_executions: usize, window: Duration) -> Self {
108        let refill_rate = max_executions as f64 / window.as_secs_f64();
109        
110        Self {
111            max_executions,
112            window,
113            semaphore: Arc::new(Semaphore::new(max_executions)),
114            token_bucket: Arc::new(Mutex::new(TokenBucket::new(
115                max_executions as f64,
116                refill_rate,
117            ))),
118        }
119    }
120    
121    /// Acquire a permit for execution
122    pub async fn acquire(&self) -> Result<RateLimitPermit, RateLimitError> {
123        // Try to acquire immediately
124        let mut bucket = self.token_bucket.lock().await;
125        
126        if bucket.try_consume(1.0) {
127            let permit = self.semaphore.clone().acquire_owned().await
128                .map_err(|_| RateLimitError::SemaphoreClosed)?;
129            return Ok(RateLimitPermit {
130                _permit: permit,
131            });
132        }
133        
134        // Calculate wait time
135        let wait_time = bucket.wait_time(1.0);
136        drop(bucket);
137        
138        // Wait and retry
139        time::sleep(wait_time).await;
140        
141        let mut bucket = self.token_bucket.lock().await;
142        if bucket.try_consume(1.0) {
143            let permit = self.semaphore.clone().acquire_owned().await
144                .map_err(|_| RateLimitError::SemaphoreClosed)?;
145            Ok(RateLimitPermit {
146                _permit: permit,
147            })
148        } else {
149            Err(RateLimitError::RateLimitExceeded(
150                "Unable to acquire rate limit permit".to_string()
151            ))
152        }
153    }
154    
155    /// Try to acquire a permit without waiting
156    pub async fn try_acquire(&self) -> Result<RateLimitPermit, RateLimitError> {
157        let mut bucket = self.token_bucket.lock().await;
158        
159        if bucket.try_consume(1.0) {
160            let permit = self.semaphore.clone().acquire_owned().await
161                .map_err(|_| RateLimitError::SemaphoreClosed)?;
162            Ok(RateLimitPermit {
163                _permit: permit,
164            })
165        } else {
166            Err(RateLimitError::RateLimitExceeded(
167                format!("Rate limit exceeded: {}/{} per {:?}", 
168                    0, self.max_executions, self.window)
169            ))
170        }
171    }
172    
173    /// Acquire with timeout
174    pub async fn acquire_timeout(&self, timeout: Duration) -> Result<RateLimitPermit, RateLimitError> {
175        match time::timeout(timeout, self.acquire()).await {
176            Ok(result) => result,
177            Err(_) => Err(RateLimitError::Timeout),
178        }
179    }
180}

182/// Permit for rate-limited execution
183pub struct RateLimitPermit {
184    _permit: tokio::sync::OwnedSemaphorePermit,
185}

187/// Adaptive rate limiter that adjusts based on system load
188pub struct AdaptiveRateLimiter {
189    /// Base rate limiter
190    base_limiter: RateLimiter,
191    
192    /// Load threshold for reducing rate
193    load_threshold: f64,
194    
195    /// Current load factor (0.0 to 1.0)
196    load_factor: Arc<Mutex<f64>>,
197}
198
199impl AdaptiveRateLimiter {
200    /// Create a new adaptive rate limiter
201    pub fn new(max_executions: usize, window: Duration, load_threshold: f64) -> Self {
202        Self {
203            base_limiter: RateLimiter::new(max_executions, window),
204            load_threshold,
205            load_factor: Arc::new(Mutex::new(0.0)),
206        }
207    }
208    
209    /// Update the current load factor
210    pub async fn update_load(&self, load: f64) {
211        let mut factor = self.load_factor.lock().await;
212        *factor = load.clamp(0.0, 1.0);
213    }
214    
215    /// Acquire a permit with adaptive rate limiting
216    pub async fn acquire(&self) -> Result<RateLimitPermit, RateLimitError> {
217        let load = *self.load_factor.lock().await;
218        
219        // If load is high, add extra delay
220        if load > self.load_threshold {
221            let extra_delay = Duration::from_millis(
222                ((load - self.load_threshold) * 1000.0) as u64
223            );
224            time::sleep(extra_delay).await;
225        }
226        
227        self.base_limiter.acquire().await
228    }
229}

235    #[tokio::test]
236    async fn test_rate_limiter_basic() {
237        let limiter = RateLimiter::new(5, Duration::from_secs(1));
238        
239        // Should be able to acquire 5 permits immediately
240        for _ in 0..5 {
241            assert!(limiter.try_acquire().await.is_ok());
242        }
243        
244        // 6th should fail
245        assert!(limiter.try_acquire().await.is_err());
246    }
247    
248    #[tokio::test]
249    async fn test_rate_limiter_refill() {
250        let limiter = RateLimiter::new(2, Duration::from_secs(1));
251        
252        // Consume all tokens
253        let _p1 = limiter.acquire().await.unwrap();
254        let _p2 = limiter.acquire().await.unwrap();
255        
256        // Should fail immediately
257        assert!(limiter.try_acquire().await.is_err());
258        
259        // Wait for refill
260        tokio::time::sleep(Duration::from_secs(1)).await;
261        
262        // Should succeed after refill
263        assert!(limiter.try_acquire().await.is_ok());
264    }

266    #[tokio::test]
267    async fn test_adaptive_rate_limiter() {
268        let limiter = AdaptiveRateLimiter::new(5, Duration::from_secs(1), 0.7);
269        
270        // Normal load
271        limiter.update_load(0.5).await;
272        assert!(limiter.acquire().await.is_ok());
273        
274        // High load
275        limiter.update_load(0.9).await;
276        let start = Instant::now();
277        assert!(limiter.acquire().await.is_ok());
278        
279        // Should have added extra delay
280        assert!(start.elapsed() >= Duration::from_millis(200));
